导入物料供应商：
INSERT INTO VDM_VENDOR_MATERIAL(
  OBJECT_RRN        ,
  ORG_RRN           ,
  IS_ACTIVE         ,
  CREATED           ,
  CREATED_BY        ,
  UPDATED           ,
  UPDATED_BY        ,
  LOCK_VERSION      ,
  VENDOR_RRN        ,
  MATERIAL_RRN      ,
  FEED_TYPE         ,
  ASSESSMENT_MARK   ,
  REFERENCED_PRICE  ,
  HIGHEST_PRICE     ,
  LAST_PRICE        ,
  LEAD_TIME         ,
  FEED_MATERIAL     ,
  LEAST_QUANTITY    ,
  INCREASE_QUANTITY ,
  COMMENTS          ,
  VENDOR_ID         ,
  MATERIAL_ID       ,
  IS_PRIMARY        ,
  PURCHASER         ,
  ADVANCE_RATIO 
)
SELECT 
  OBJECT_RRN.NEXTVAL , 
  2501932       ,
  M.IS_ACTIVE     ,
  M.CREATED           ,
  M.CREATED_BY        ,
  M.UPDATED           ,
  M.UPDATED_BY        ,
  M.LOCK_VERSION      ,
  V.OBJECT_RRN        ,
  P.OBJECT_RRN      ,
  FEED_TYPE         ,
  ASSESSMENT_MARK   ,
  REFERENCED_PRICE  ,
  HIGHEST_PRICE     ,
  LAST_PRICE        ,
  LEAD_TIME         ,
  FEED_MATERIAL     ,
  LEAST_QUANTITY    ,
  INCREASE_QUANTITY ,
  M.COMMENTS          ,
  V.VENDOR_ID         ,
  P.MATERIAL_ID       ,
  IS_PRIMARY        ,
  PURCHASER         ,
  ADVANCE_RATIO 
FROM 
(SELECT M.*, V.VENDOR_ID V_VENDOR_ID, P.MATERIAL_ID P_MATERIAL_ID
FROM (SELECT * FROM VDM_VENDOR_MATERIAL M WHERE M.ORG_RRN = 139420) M, VDM_VENDOR V, PDM_MATERIAL P
WHERE M.VENDOR_RRN = V.OBJECT_RRN
  AND M.MATERIAL_RRN = P.OBJECT_RRN) M,
(SELECT * FROM VDM_VENDOR V WHERE V.ORG_RRN = '2501932') V,
(SELECT * FROM PDM_MATERIAL P WHERE P.ORG_RRN = '2501932') P
WHERE M.V_VENDOR_ID = V.VENDOR_ID
  AND M.P_MATERIAL_ID = P.MATERIAL_ID
